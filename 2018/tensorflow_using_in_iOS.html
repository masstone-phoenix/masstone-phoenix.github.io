全部编译

在makefile中找到下面的片断，将ANDROID_TYPES 设置为D__ANDROID_TYPES_FULL__。

# If ANDROID_TYPES is not set assume __ANDROID_TYPES_SLIM__
# ifeq ($(ANDROID_TYPES),)
#     ANDROID_TYPES := -D__ANDROID_TYPES_SLIM__
# endif
ANDROID_TYPES := -D__ANDROID_TYPES_FULL__
